CREATE DATABASE ESHOPPER;
CREATE TABLE IKLAN(
	ID_PRODUK varchar(16) NOT NULL PRIMARY KEY,
	NAMA_PRODUK varchar(60),
	HARGA smallint(6),
	JUMLAH smallint(4),
	KATEGORI varchar(15),
	DESKRIPSI varchar(100),
	ID_MEMBER int(11) 
);

CREATE TABLE MEMBER(
 ID int(11) NOT NULL PRIMARY KEY AUTO_INCREMENT,
 NAMA varchar(60) NOT NULL,
 EMAIL varchar(30) NOT NULL,
 NOHP varchar(20) NOT NULL,
 PASSWD char(32) NOT NULL
)AUTO_INCREMENT=13247;
